== docker build .  # docker build command is used to create Docker images from a Dockerfile and a "context".
== docker image ls # docker image ls command is used to list all the Docker images stored locally on your system.
== docker run [Image ID] # This command creates and starts a new container from the image with the ID e9f50c18a0e6.
== docker run -p -d 3000:3000 [Image ID] # This command is used for porting mapping & -d flag is used for detached mode.
== docker ps # This command lists only the running containers
== docker ps -a # This command is used to list all containers (running and non-running)
== docker rm [ID or container name] # This command is used to remove the container
